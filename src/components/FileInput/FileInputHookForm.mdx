# \<FileInput.HookForm />

FileInput.HookForm is a `react-hook-form` support version of FileInput. \

<Canvas of={Stories.HookForm} />

### Usage

```tsx
import { useForm, FormProvider } from "react-hook-form"
import { FileInput } from "@reactleaf/input"

function FormComponent() {
  const form = useForm()

  return (
    <FormProvider {...form}>
      <FileInput.HookForm name="image" label="Image" maxFileSize={10 * 1000 * 1000} />
    </FormProvider>
  )
}
```
